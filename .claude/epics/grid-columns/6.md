### NumberInput.tsx
Spec en `docs/reference/technical-decisions.md` → Sección 9.5.

```tsx
interface NumberInputProps {
  label: string;
  value: number;
  onChange: (value: number) => void;
  min?: number;
  max?: number;
  step?: number;
  suffix?: string;   // 'px', '%'

```

Usar `<sp-textfield type="number">` de Adobe Spectrum. El `suffix` se muestra como texto decorativo junto al input.

**Accesibilidad:** el `label` debe estar vinculado al input con `htmlFor` / `id` únicos (usar `useId()` de React 18).

### ApplyButton.tsx
Botón principal de aplicación. Connecto directamente con stores y servicios.

```tsx
// No recibe props — lee el estado de los stores directamente
export function ApplyButton() {
  const config = useGridStore((s) => s.config);
  const { document } = useDocument();
  const { isApplying, setApplying, setError, setSuccess } = useUIStore();

  const handleApply = async () => {
    if (!document) return;
    setApplying(true);
    try {
      const guides = generateGuides(config, {
        width: document.width,
        height: document.height,
      });
      await photoshopBridge.applyGuides(guides, config.applyMode);
      setSuccess(true);
    } catch (err) {
      setError(err instanceof Error ? err.message : 'Error al aplicar el grid');
    } finally {
      setApplying(false);
    }
  };

  return (
    <sp-button
      variant="cta"
      disabled={!document || isApplying || undefined}
      onClick={handleApply}
    >
      {isApplying ? 'Aplicando...' : config.applyMode === 'replace' ? 'Reemplazar guías' : 'Agregar guías'}
    </sp-button>
  );
}
```

**Nota:** El label del botón cambia según el modo (`applyMode`): "Reemplazar guías" o "Agregar guías".
